func (t *AuthUseCase) AuthHandler(ctx context.Context) (context.Context, error) {
	token, err := grpc_auth.AuthFromMD(ctx, "bearer")
	if err != nil {
		return nil, err
	}

	authUser, err := t.authenticationClient.VerifyToken(token)
	if err != nil {
		return nil, err
	}

	serverTransportStream := grpc.ServerTransportStreamFromContext(ctx)
	pass, err := t.authorizationClient.EnforceRBAC(authUser.UserID, authUser.TenantID, serverTransportStream.Method())
	if err != nil {
		return nil, status.Errorf(codes.Unauthenticated, "Enforce error: %v", err)
	}
	if !pass {
		return nil, ErrUnauthorized
	}

	ctx = context.WithValue(ctx, UserIDKey, authUser.UserID)
	ctx = context.WithValue(ctx, TenantIDKey, authUser.TenantID)

	return ctx, nil
}
